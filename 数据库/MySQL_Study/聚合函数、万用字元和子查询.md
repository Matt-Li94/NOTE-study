```cpp

-- 1.聚合函数:count max min avg sum
select count(*) from `employee`; 
select max(`money`) from `employee`;
select min(`money`) from `employee`;
select avg(`money`) from `employee`;
select sum(`money`) from `employee`;

-- 万用子元 % 和 _ 前者可以表示多个 后者只能表示一个

-- 取得电话号码尾号是335的客户
select * from `client` where `phone` like '%335'; 
-- 取得姓艾的客户的信息
select * from `client` where `name` like '艾%'; 
-- 取得生日在12月客户的信息
select * from `client` where `birth_day` like '%-12-%'; 

-- 子查询 在一个select里面嵌套另一个select 
-- 找到研发部门的经理的名字
-- 使用in而不是 = 因为内查询有可能出来不止一个值 
select `name` from `employee` where `employ_id` in (
	select `manager_id` from `branch` where `branch_name` = '研发' 
);
```
