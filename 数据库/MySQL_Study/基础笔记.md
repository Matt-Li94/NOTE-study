```cpp
// MySQL里面常用的变量
int -- 整数
decimal(n.m) -- 有小数点的数，其中n代表全部位数，m代表小数点的位数
varchar(n) --字符串,其中n代表最多存放的单词
blob --（Binary Large Object）二进制大对象eg图片 影片 档案
data -- 'YYYY-MM-DD' 日期
timestamp -- 'YYYY-MM-DD HH:MM:SS' 时间


create database `lbh_database`; //创建数据库
show databases; //展示数据库
drop database ·lbh_database·; // 删除特定的数据库
use `lbh_database`; //使用特定的数据库 后面的操作都在这里进行
create table `students`( //创建表格
    `student_id` int primary key, //主键
    `name` varchar(30) not null, 
    `major` varchar(30) unique,
    `name_id` int unique auto_increment, //自增且unique
    foreign key (`name`) references `teacher`(teacher_name) on delete set null 
    //设置外键而且如果外键被删除了就把这个 外键设置为null
    foreign key (`name`) references `teacher`(teacher_name) on delete cascade
    //设置外键而且如果外键被删除了就把这个整一条信息都删除掉
    // 外键一般是其他表格的主键 当然不是主键的话就要是具有unique属性的键了
);

describe `students`; //展示表格的属性
drop table `students`; //删除表格

alter table `students` add `gpa` decimal(3,2); //添加表格的属性
alter table `students` drop  column `gpa`; //删除表格属性

insert into `students` value(1,"李彬泓",'电子信息');
insert into `students` value(2,'伞兵一号','卡布奇诺');
insert into `students`(`name`,`student_id`,`major`) value('伞兵二号',3,'我没开挂'); //添加表格的每一行信息

select * from `students`; //展示表格的所有信息
```
