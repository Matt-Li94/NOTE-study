# 数组最长连续子序列长度
灵神的代码很牛逼

```cpp
class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        // 这个做不了动态规划 只能使用哈希表的做法去做
        unordered_set<int> uset(nums.begin(), nums.end());
        // 遍历set里面的每一个元素
        int length{0};
        for(const auto& i : uset) {
            if(uset.find(i - 1) != uset.end()) continue; 
            // 找到了i-1 那么最大值必不在i开始 因为i - 1绝对比 i开始的要长
            else {
            // 找不到i - 1 从i开始搜索
                int inext = i + 1;
                int len{1};
                while(uset.find(inext) != uset.end()) {
                    len++;
                    inext++;
                }
                length = max(length, len);
            }
        }
        return length;
    }
};
```
