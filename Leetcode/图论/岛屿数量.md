# 岛屿数量
这道和可达路径是深搜的入门级题目 
思路是

1.dfs的作用是遍历当前陆地所在的这个岛屿并把它们全部插满旗帜
2.之后对每一个节点启动dfs搜索 对这个节点所在的陆地岛屿插满旗帜 之后计数值加一
3.他没有回溯的一个东西
代码如下：

```cpp
class Solution {
private:
    int four_dir[4][2] = {1,0, 0,1, -1,0, 0,-1}; // 学习新的知识 方向向量
    int count = 0;
public:
    void dfs(vector<vector<char>>& grid,vector<vector<bool>>& visited, int x, int y) {
        if(x < 0 || x >= grid.size() || y <0 || y >=grid[0].size() || grid[x][y] == '0' || visited[x][y] == true) return ;
        visited[x][y] = true;
        for(int i{0};i<4;i++) {
            int nextx = x + four_dir[i][0];
            int nexty = y + four_dir[i][1];
            dfs(grid,visited,nextx,nexty);
        }
    }
    int numIslands(vector<vector<char>>& grid) {
        int row = grid.size();
        int col = grid[0].size();
        vector<vector<bool>> visited(row, vector<bool>(col, false));
        for(int i{0};i<row;i++) {
            for(int j{0};j<col;j++) {
                if(grid[i][j] == '1' && visited[i][j] == false) {
                    dfs(grid,visited,i,j);
                    count++;
                }
            }
        }
        return count;
    }
};
```
