# 和为k的子数组

前缀和典型例题

注意先计算count再进入哈希

```cpp
class Solution {
private:
    unordered_map<int, int> umap;
public:
    int subarraySum(vector<int>& nums, int k) {
        // 采用前缀和的做法 注意先计算count再进入哈希
        // presum2 - presum1 = k -----> presum2 - k = presum1.
        umap[0] = 1;
        int presum{0};
        int count{0};
        for(const int& i : nums) {
            presum = i + presum;
            count = count + umap[presum - k];
            umap[presum]++;
        }
        return count;
    }
};
```
