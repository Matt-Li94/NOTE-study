# 消失的数字

分享一个奇淫技巧 就是遍历数组上面的数字 把对应的索引值上面的数字变成负数
那么第二次遍历的时候发现正数的说明这个数字的下标没有出现过了

```cpp
class Solution {
public:
    vector<int> findDisappearedNumbers(vector<int>& nums) {
        // 还需要在哈希表上面进行空间的优化
        // 使用数组来进行一个标记
        // 把数组对应的索引上面的数组变成负数 如果它是正数的话才变 
        vector<int> result;
        for(const int& i : nums) {
            int index = abs(i) - 1;
            if(nums[index] > 0) nums[index] = -nums[index];
        }
        for(int i{0};i<nums.size();i++) {
            if(nums[i] >= 0) result.push_back(i + 1);
            // 如果是正数说明这个数字对应的下标的数字没有出现过 因为出现过了那么这个数字就会变成负数了
        }
        return result;
    }
};
```
