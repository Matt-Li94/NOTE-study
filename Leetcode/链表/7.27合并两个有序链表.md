# 合并两个有序链表
创建一个新链表节点 比较两个链表 哪一个小就把哪一个添加到新链表里面 然后更新这三个指针
```cpp
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        // 创建哨兵节点
        // 比较list1和list2 哪一个小就把哪一个新链表的指针指向它 并且更新list1 和 2
        // 最后剩余的那个链表直接拼接到新链表后面就行了
        ListNode* dummy = new ListNode(0);
        ListNode* cur = dummy;
        while(list1 != nullptr && list2 != nullptr) {
            if(list1->val > list2->val) {
                cur->next = list2;
                list2 = list2->next;
            }
            else {
                cur->next = list1;
                list1 = list1->next;
            }
            cur = cur->next;
        }
        if(list1 != nullptr) cur->next = list1;
        if(list2 != nullptr) cur->next = list2;
        return dummy->next;
    }
};
```
