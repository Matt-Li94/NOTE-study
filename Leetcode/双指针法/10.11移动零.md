# 移动零
这道题目也是典型使用双指针的一个题目
思路就是 快慢指针 当快指针遇非0的时候就交换快慢指针的一个数值 同时快慢指针加1
```cpp
class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        // 双指针法 快慢指针 当快指针遇非0的时候就交换快慢指针的一个数值 同时快慢指针加1
        if(nums.size() == 1) return ;
        int fast{0};
        int slow{0};
        while(fast < nums.size() && slow < nums.size()) {
            while(nums[fast] == 0 && fast < nums.size() - 1) {
                fast++;
            }
            cout<<"fast is:"<<fast<<" ";
            swap(nums[slow], nums[fast]);
            slow++;
            fast++;
        }
    }
};
```
