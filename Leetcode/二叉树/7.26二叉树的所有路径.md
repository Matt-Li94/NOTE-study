# 获取二叉树的所有路径
这道题目有点乱拳打死老师傅的感觉了 其实我也不是很清楚为什么能够做出来 但是好像就是这个道路
思路写在代码里面了
```cpp
class Solution {
public:
    // 乱拳打死老师傅
    // 思路是使用一个变量存当前节点的数值 之后递归 递归的时候复制这个节点 因为要包含之前的路径 那么到达空节点的时候就是结束了 这个时候添加到另外一个答案变量中 完美解决问题
    void traversal(TreeNode* root,string sPath,vector<string>& result) {
        if(root->left == nullptr && root->right == nullptr) {// 叶子节点结束 这里是添加路径的逻辑
            sPath = sPath + to_string(root->val);
            result.push_back(sPath);
            return ;
        }
        sPath = sPath + to_string(root->val) + "->"; //添加当前节点数值
        if(root->left != nullptr) traversal(root->left,sPath,result); //递归 这里你使用了值传递sPath，那么每一个递归栈里面都有独立的sPath 并且也完成了复制的操作了 
        if(root->right != nullptr) traversal(root->right,sPath,result);
    }
    vector<string> binaryTreePaths(TreeNode* root) {
        vector<string> result;
        traversal(root,"",result);
        return result;
    }
};
```
