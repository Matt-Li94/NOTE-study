# 二叉树的最小深度
还是比较习惯使用自顶向下遍历的思路去做递归
这里和求最大深度的逻辑差不多 只不过 mindepth初始化为INT_MAX。
而且只有在叶子节点才更新mindepth 不然会被一个非叶子节点的较小值卡着无法正确更新
不难写代码如下：
```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void travesar(TreeNode* root,int cur_depth,int& min_depth){
        if(root == nullptr) return;
        cur_depth++;
        if(root->left == nullptr && root->right == nullptr){
            min_depth = min(cur_depth,min_depth);
        }
        travesar(root->left,cur_depth,min_depth);
        travesar(root->right,cur_depth,min_depth);
    }
    int minDepth(TreeNode* root) {
        if(root == nullptr) return 0;
        int min_depth = INT_MAX;
        travesar(root,0,min_depth);
        return min_depth;
    }
};
```
